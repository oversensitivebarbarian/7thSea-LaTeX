%%%% NICETABLE %%%%
% Uses tabularray — no colortbl/rowcolors, no TeX Live 2025 issues.
% Usage:
%   \begin{NiceTable}{colspec}                        — plain table
	%   \begin{NiceTable}[My Header]{colspec}             — with header
		%   \begin{NiceTable}[My Header][blue!20]{colspec}    — header + row color
			
			\NewDocumentEnvironment{NiceTable}{o O{white} m}{%
				\par\vspace{9pt plus 3pt minus 3pt}\noindent
				\IfValueT{#1}{{\large\bfseries\sffamily #1\nopagebreak}\par\vspace{5pt plus 2pt minus 2pt}\noindent}%
				\small\sffamily
				\begin{tblr}{
						colspec={#3},
						row{odd}={#2},
						row{even}={},
						rowsep=3pt,
						row{1}={abovesep=8pt, belowsep=8pt},
					}%
				}{%
				\end{tblr}%
				\vspace{9pt plus 3pt minus 3pt}%
			}
			
			%%%% CITATION %%%%
			\NewDocumentEnvironment{Citation}{o m}{%
				\par\vspace{1em}
				\begin{center}\itshape ``#2''\end{center}
				\IfValueT{#1}{\par\noindent\hfill{\itshape\textemdash\ #1}\par}
				\vspace{0.5em}
			}{}

			\newcommand{\drawcoverimage}[4]{%
				\savebox{\overlaytempbox}{\includegraphics[width=#2]{#1}}%
				\ifdim\ht\overlaytempbox<#3%
				\savebox{\overlaytempbox}{\includegraphics[height=#3]{#1}}%
				\fi
				\node[#4]{\usebox{\overlaytempbox}};%
			}
			
			\NewDocumentEnvironment{SpecialNotes}{O{} m +b}{%
				\edef\overlaypath{#1}%
				\savebox{\specialnotesbox}{%
					\begin{minipage}{\columnwidth}
						\begin{mdframed}[
							linewidth=0pt,
							innerleftmargin=0pt, innerrightmargin=5pt,
							innertopmargin=6pt,  innerbottommargin=6pt,
							backgroundcolor=none, skipabove=0pt, skipbelow=0pt]
							\begin{minipage}{\dimexpr\columnwidth-32pt\relax}
								\begin{center}
									{\color{specialNotesTextColor}\Large\MakeUppercase{#2}\par}
								\end{center}
								\normalsize
								\begin{center}
									\begin{tikzpicture}
										\draw[specialNotesTextColor, dotted, line width=6pt,
										dash pattern=on 1pt off 2pt]
										(0,0) -- (\columnwidth,0);
									\end{tikzpicture}
								\end{center}
								\setlength{\parskip}{1em}
								\color{specialNotesTextColor}
								#3
								\vspace{2em}
							\end{minipage}
						\end{mdframed}
					\end{minipage}%
				}%
				\setlength{\specialnotesheight}{\dimexpr\ht\specialnotesbox+\dp\specialnotesbox\relax}%
				\setlength{\specialnotestotalheight}{\dimexpr\mytopmargin+\specialnotesheight\relax}%
				\begin{tikzpicture}[remember picture, overlay]
					\node[inner sep=0pt, outer sep=0pt, anchor=north west]
					(specialnotesanchor) at (0,0) {};
				\end{tikzpicture}
				\begin{tikzpicture}[remember picture, overlay]
					\fill[baseColor]
					([xshift=\specialnotesxshift]specialnotesanchor.north west |- current page.north)
					rectangle ++(\columnwidth,-\specialnotestotalheight);
					\ifx\overlaypath\empty\else
					\begin{scope}
						\clip
						([xshift=\specialnotesxshift]specialnotesanchor.north west |- current page.north)
						rectangle ++(\columnwidth,-\specialnotestotalheight);
						\drawcoverimage{\overlaypath}{\columnwidth}{\specialnotestotalheight}{%
							opacity=0.15, inner sep=0pt, outer sep=0pt, anchor=north,
							at={([xshift=\specialnotesxshift+0.5\columnwidth]
								specialnotesanchor.north west |- current page.north)}%
						}
					\end{scope}
					\fi
				\end{tikzpicture}
				\noindent\usebox{\specialnotesbox}\par
				\vspace{2\baselineskip}
			}{}
